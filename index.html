Description: Given some text with html tags, return this text with replacement unmark list's items for items with 'li':

                                    
function modifyLi(text) {
    const regP = /\/(p|li)><p>(-|·)\s\D.+[,.;:]<\/p>/s

    while (regP.test(text)) {
        let foundP =  text.match(regP)

        let handledP = foundP[0]
            .replaceAll('</p>', '</li>')
            .replaceAll('<p>- ', '<li>')
            .replaceAll('<p>· ', '<li>')
        text = text.replaceAll(foundP[0], handledP)
    }
    
    return text
}
